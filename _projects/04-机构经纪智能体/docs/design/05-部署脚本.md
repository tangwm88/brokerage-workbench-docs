## 二、Docker Compose配置

### 2.1 主配置文件

```yaml
# docker-compose.yml
version: '3.8'

services:
  # Nginx反向代理
  nginx:
    image: nginx:alpine
    container_name: brokerage-nginx
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf:ro
      - ./nginx/conf.d:/etc/nginx/conf.d:ro
      - ./nginx/ssl:/etc/nginx/ssl:ro
      - ./h5/dist:/usr/share/nginx/html/h5:ro
      - nginx_logs:/var/log/nginx
    depends_on:
      - backend
      - openclaw
    networks:
      - brokerage-network
    restart: unless-stopped

  # OpenClaw Gateway
  openclaw:
    image: openclaw/gateway:latest
    container_name: brokerage-openclaw
    ports:
      - "8080:8080"
    volumes:
      - ./openclaw/config.yaml:/app/config.yaml:ro
      - ./openclaw/skills:/app/skills:ro
      - openclaw_data:/app/data
    environment:
      - DATABASE_URL=postgresql://postgres:${DB_PASSWORD}@postgres:5432/openclaw
      - REDIS_URL=redis://redis:6379/0
      - BASE_API_URL=${BASE_API_URL}
      - BASE_APP_ID=${BASE_APP_ID}
      - BASE_APP_SECRET=${BASE_APP_SECRET}
    depends_on:
      - postgres
      - redis
    networks:
      - brokerage-network
    restart: unless-stopped

  # 后端API服务
  backend:
    image: brokerage/backend:latest
    container_name: brokerage-backend
    build:
      context: ./backend
      dockerfile: Dockerfile
    ports:
      - "3000:3000"
    environment:
      - NODE_ENV=production
      - PORT=3000
      - DATABASE_URL=postgresql://postgres:${DB_PASSWORD}@postgres:5432/brokerage
      - REDIS_URL=redis://redis:6379/1
      - BASE_API_URL=${BASE_API_URL}
      - BASE_APP_ID=${BASE_APP_ID}
      - BASE_APP_SECRET=${BASE_APP_SECRET}
      - JWT_SECRET=${JWT_SECRET}
      - OPENCLAW_URL=http://openclaw:8080
    depends_on:
      - postgres
      - redis
    networks:
      - brokerage-network
    restart: unless-stopped
    deploy:
      replicas: 2
      resources:
        limits:
          cpus: '1.0'
          memory: 1G
        reservations:
          cpus: '0.5'
          memory: 512M

  # PostgreSQL数据库
  postgres:
    image: postgres:15-alpine
    container_name: brokerage-postgres
    ports:
      - "5432:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./postgres/init:/docker-entrypoint-initdb.d:ro
    environment:
      - POSTGRES_USER=postgres
      - POSTGRES_PASSWORD=${DB_PASSWORD}
      - POSTGRES_DB=brokerage
    networks:
      - brokerage-network
    restart: unless-stopped
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U postgres"]
      interval: 10s
      timeout: 5s
      retries: 5

  # Redis缓存
  redis:
    image: redis:7-alpine
    container_name: brokerage-redis
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
      - ./redis/redis.conf:/usr/local/etc/redis/redis.conf:ro
    command: redis-server /usr/local/etc/redis/redis.conf
    networks:
      - brokerage-network
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5

  # H5构建（仅用于构建阶段）
  h5-builder:
    image: node:20-alpine
    container_name: brokerage-h5-builder
    working_dir: /app
    volumes:
      - ./h5:/app
      - h5_dist:/app/dist
    command: sh -c "npm install && npm run build"
    profiles:
      - build

volumes:
  postgres_data:
    driver: local
  redis_data:
    driver: local
  openclaw_data:
    driver: local
  h5_dist:
    driver: local
  nginx_logs:
    driver: local

networks:
  brokerage-network:
    driver: bridge
```

## 三、Nginx配置

### 3.1 主配置文件

```nginx
# nginx/nginx.conf
user nginx;
worker_processes auto;
error_log /var/log/nginx/error.log warn;
pid /var/run/nginx.pid;

events {
    worker_connections 4096;
    use epoll;
    multi_accept on;
}

http {
    include /etc/nginx/mime.types;
    default_type application/octet-stream;

    # 日志格式
    log_format main '$remote_addr - $remote_user [$time_local] "$request" '
                    '$status $body_bytes_sent "$http_referer" '
                    '"$http_user_agent" "$http_x_forwarded_for" '
                    'rt=$request_time uct="$upstream_connect_time" '
                    'uht="$upstream_header_time" urt="$upstream_response_time"';

    access_log /var/log/nginx/access.log main;

    # 基础配置
    sendfile on;
    tcp_nopush on;
    tcp_nodelay on;
    keepalive_timeout 65;
    types_hash_max_size 2048;
    client_max_body_size 50M;

    # Gzip压缩
    gzip on;
    gzip_vary on;
    gzip_proxied any;
    gzip_comp_level 6;
    gzip_types text/plain text/css text/xml application/json 
               application/javascript application/rss+xml 
               application/atom+xml image/svg+xml;

    # 包含其他配置
    include /etc/nginx/conf.d/*.conf;
}
```

### 3.2 HTTP服务配置

```nginx
# nginx/conf.d/default.conf
server {
    listen 80;
    server_name _;
    
    # 健康检查
    location /health {
        access_log off;
        return 200 "healthy\n";
        add_header Content-Type text/plain;
    }

    # H5静态文件
    location /h5 {
        alias /usr/share/nginx/html/h5;
        index index.html;
        try_files $uri $uri/ /h5/index.html;
        
        # 缓存控制
        location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2)$ {
            expires 1y;
            add_header Cache-Control "public, immutable";
        }
        
        # HTML不缓存
        location ~* \.html$ {
            expires -1;
            add_header Cache-Control "no-cache, no-store, must-revalidate";
        }
    }

    # API代理
    location /api {
        proxy_pass http://backend:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        proxy_read_timeout 300s;
        proxy_connect_timeout 75s;
    }

    # WebSocket代理
    location /ws {
        proxy_pass http://openclaw:8080;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_read_timeout 86400s;
        proxy_send_timeout 86400s;
    }

    # OpenClaw API代理
    location /openclaw {
        proxy_pass http://openclaw:8080;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_read_timeout 300s;
    }

    # 根路径重定向到H5
    location = / {
        return 302 /h5/;
    }
}
```

### 3.3 HTTPS生产配置

```nginx
# nginx/conf.d/prod.conf
server {
    listen 80;
    server_name workbench.company.com;
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    server_name workbench.company.com;

    # SSL证书
    ssl_certificate /etc/letsencrypt/live/workbench.company.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/workbench.company.com/privkey.pem;
    ssl_trusted_certificate /etc/letsencrypt/live/workbench.company.com/chain.pem;

    # SSL配置
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384;
    ssl_prefer_server_ciphers off;
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 1d;
    ssl_session_tickets off;

    # OCSP Stapling
    ssl_stapling on;
    ssl_stapling_verify on;
    resolver 8.8.8.8 8.8.4.4 valid=300s;
    resolver_timeout 5s;

    # Security Headers
    add_header Strict-Transport-Security "max-age=63072000" always;
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Referrer-Policy "strict-origin-when-cross-origin" always;

    # H5静态文件
    location /h5 {
        alias /usr/share/nginx/html/h5;
        index index.html;
        try_files $uri $uri/ /h5/index.html;

        # 强缓存静态资源
        location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2|ttf|eot)$ {
            expires 1y;
            add_header Cache-Control "public, immutable";
            add_header Vary "Accept-Encoding";
            access_log off;
        }
    }

    # API代理（负载均衡）
    location /api {
        proxy_pass http://backend;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_read_timeout 300s;
        proxy_connect_timeout 75s;

        # 限流
        limit_req zone=api_limit burst=20 nodelay;
    }

    # WebSocket
    location /ws {
        proxy_pass http://openclaw:8080;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_read_timeout 86400s;
    }

    location = / {
        return 302 /h5/;
    }
}

# 上游服务器配置
upstream backend {
    least_conn;
    server backend:3000 max_fails=3 fail_timeout=30s;
    server backend:3000 max_fails=3 fail_timeout=30s;
    server backend:3000 max_fails=3 fail_timeout=30s;
    keepalive 32;
}
```

## 五、启动脚本

### 5.1 部署脚本

```bash
#!/bin/bash
# deploy.sh

set -e

# 颜色定义
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m'

# 配置
COMPOSE_FILE="docker-compose.yml"
ENV_FILE=".env"
BACKUP_DIR="./backups"

# 函数定义
log_info() {
    echo -e "${GREEN}[INFO]${NC} $1"
}

log_warn() {
    echo -e "${YELLOW}[WARN]${NC} $1"
}

log_error() {
    echo -e "${RED}[ERROR]${NC} $1"
}

# 检查环境
check_environment() {
    log_info "检查部署环境..."
    
    # 检查Docker
    if ! command -v docker &> /dev/null; then
        log_error "Docker未安装"
        exit 1
    fi
    
    # 检查Docker Compose
    if ! command -v docker-compose &> /dev/null; then
        log_error "Docker Compose未安装"
        exit 1
    fi
    
    # 检查.env文件
    if [ ! -f "$ENV_FILE" ]; then
        log_error ".env文件不存在，请从.env.example复制并配置"
        exit 1
    fi
    
    log_info "环境检查通过"
}

# 创建目录
create_directories() {
    log_info "创建必要目录..."
    
    mkdir -p \
        ./nginx/ssl \
        ./nginx/logs \
        ./postgres/init \
        ./postgres/data \
        ./redis/data \
        ./openclaw/skills \
        ./openclaw/data \
        ./h5/dist \
        ./backups
    
    log_info "目录创建完成"
}

# 构建H5
build_h5() {
    log_info "构建H5前端..."
    
    cd h5
    npm install
    npm run build
    cd ..
    
    log_info "H5构建完成"
}

# 数据库迁移
migrate_database() {
    log_info "执行数据库迁移..."
    
    # 等待PostgreSQL启动
    sleep 5
    
    docker-compose exec -T backend npm run migrate
    
    log_info "数据库迁移完成"
}

# 备份数据
backup_data() {
    if [ -d "./postgres/data" ] && [ "$(ls -A ./postgres/data)" ]; then
        log_info "备份现有数据..."
        
        BACKUP_FILE="$BACKUP_DIR/backup_$(date +%Y%m%d_%H%M%S).sql"
        docker-compose exec -T postgres pg_dump -U postgres brokerage > "$BACKUP_FILE"
        gzip "$BACKUP_FILE"
        
        log_info "数据备份完成: $BACKUP_FILE.gz"
    fi
}

# 主部署流程
deploy() {
    log_info "开始部署机构经纪工作台..."
    
    check_environment
    create_directories
    
    # 备份数据
    backup_data
    
    # 构建H5
    build_h5
    
    # 拉取最新镜像
    log_info "拉取Docker镜像..."
    docker-compose pull
    
    # 启动服务
    log_info "启动服务..."
    docker-compose up -d
    
    # 等待服务启动
    log_info "等待服务启动..."
    sleep 10
    
    # 数据库迁移
    migrate_database
    
    # 健康检查
    log_info "执行健康检查..."
    if curl -f http://localhost/health > /dev/null 2>&1; then
        log_info "✅ 部署成功！"
        log_info "访问地址: http://localhost/h5/"
    else
        log_error "❌ 健康检查失败，请检查日志"
        docker-compose logs
        exit 1
    fi
}

# 查看状态
status() {
    docker-compose ps
}

# 查看日志
logs() {
    docker-compose logs -f "$1"
}

# 停止服务
stop() {
    log_info "停止服务..."
    docker-compose down
}

# 重启服务
restart() {
    log_info "重启服务..."
    docker-compose restart
}

# 更新部署
update() {
    log_info "更新部署..."
    
    backup_data
    docker-compose pull
    docker-compose up -d
    migrate_database
    
    log_info "更新完成"
}

# 清理数据
clean() {
    log_warn "警告: 这将删除所有数据！"
    read -p "确定要继续吗? (yes/no): " confirm
    
    if [ "$confirm" = "yes" ]; then
        docker-compose down -v
        rm -rf ./postgres/data/*
        rm -rf ./redis/data/*
        log_info "数据已清理"
    else
        log_info "操作已取消"
    fi
}

# 使用说明
usage() {
    echo "使用方法: $0 {deploy|status|logs|stop|restart|update|clean}"
    echo ""
    echo "命令:"
    echo "  deploy   - 完整部署"
    echo "  status   - 查看服务状态"
    echo "  logs     - 查看日志 (可指定服务名)"
    echo "  stop     - 停止服务"
    echo "  restart  - 重启服务"
    echo "  update   - 更新部署"
    echo "  clean    - 清理数据"
}

# 主逻辑
case "${1:-deploy}" in
    deploy)
        deploy
        ;;
    status)
        status
        ;;
    logs)
        logs "$2"
        ;;
    stop)
        stop
        ;;
    restart)
        restart
        ;;
    update)
        update
        ;;
    clean)
        clean
        ;;
    *)
        usage
        exit 1
        ;;
esac
```

```bash
#!/bin/bash
# logs-collect.sh

docker-compose logs --no-color > "$OUTPUT_DIR/docker-compose.log" 2>&1

docker-compose version > "$OUTPUT_DIR/docker-compose-version.txt"
docker system df > "$OUTPUT_DIR/docker-system-df.txt"
docker ps -a > "$OUTPUT_DIR/docker-ps.txt"

echo "打包..."
tar -czf "$OUTPUT_DIR.tar.gz" -C "$OUTPUT_DIR" .
rm -rf "$OUTPUT_DIR"

echo "日志收集完成: $OUTPUT_DIR.tar.gz"
```

### 8.1 数据库备份脚本

```bash
#!/bin/bash
# backup.sh

set -e

BACKUP_DIR="./backups"
DB_NAME="brokerage"
DB_USER="postgres"
RETENTION_DAYS=30

mkdir -p "$BACKUP_DIR"

# 创建备份
BACKUP_FILE="$BACKUP_DIR/${DB_NAME}_$(date +%Y%m%d_%H%M%S).sql"

echo "开始备份数据库..."
docker-compose exec -T postgres pg_dump -U $DB_USER $DB_NAME > "$BACKUP_FILE"

# 压缩
gzip "$BACKUP_FILE"
echo "备份完成: ${BACKUP_FILE}.gz"

# 清理过期备份
echo "清理过期备份..."
find "$BACKUP_DIR" -name "${DB_NAME}_*.sql.gz" -mtime +$RETENTION_DAYS -delete

echo "备份任务完成"
```

### 8.2 数据库恢复脚本

```bash
#!/bin/bash
# restore.sh

set -e

if [ -z "$1" ]; then
    echo "使用方法: $0 <备份文件>"
    echo "可用备份:"
    ls -lh ./backups/*.sql.gz 2>/dev/null || echo "无备份文件"
    exit 1
fi

BACKUP_FILE="$1"
DB_NAME="brokerage"
DB_USER="postgres"

if [ ! -f "$BACKUP_FILE" ]; then
    echo "备份文件不存在: $BACKUP_FILE"
    exit 1
fi

echo "警告: 这将覆盖现有数据库！"
read -p "确定要继续吗? (yes/no): " confirm

if [ "$confirm" != "yes" ]; then
    echo "操作已取消"
    exit 0
fi

# 停止相关服务
echo "停止相关服务..."
docker-compose stop backend openclaw

# 恢复数据库
echo "恢复数据库..."
if [[ "$BACKUP_FILE" == *.gz ]]; then
    gunzip -c "$BACKUP_FILE" | docker-compose exec -T postgres psql -U $DB_USER $DB_NAME
else
    docker-compose exec -T postgres psql -U $DB_USER $DB_NAME < "$BACKUP_FILE"
fi

# 启动服务
echo "启动服务..."
docker-compose start backend openclaw

echo "恢复完成"
```
