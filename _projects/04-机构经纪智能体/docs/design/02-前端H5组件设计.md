```typescript
  history: createWebHistory(import.meta.env.BASE_URL),
  routes

    // 在实际环境中跳转到登录页
    // next('/login')
    next()
  } else {
    next()
  }
})
```

### 3.2 机器人对话页 (Chat)

```vue
<!-- src/views/chat/Index.vue -->
<template>
  <div class="chat-page">
    <!-- 顶部导航 -->
    <van-nav-bar
      title="机构经纪数字员工"
      left-arrow
      @click-left="goBack"
    />
    
    <!-- 消息列表 -->
    <div class="chat-container" ref="chatContainer">
      <div class="message-list">
        <div 
          v-for="(msg, index) in messages" 
          :key="index"
          class="message-item"
          :class="{ 'self': msg.isSelf }"
        >
          <ChatBubble 
            :message="msg"
            @action="handleMessageAction"
          />
        </div>
      </div>
    </div>
    
    <!-- 快捷提示词 -->
    <QuickPrompts 
      :prompts="quickPrompts"
      @select="handleQuickPrompt"
    />
    
    <!-- 输入框 -->
    <ChatInput 
      v-model="inputMessage"
      :loading="sending"
      @send="sendMessage"
      @voice="handleVoiceInput"
    />
  </div>
</template>

<script setup lang="ts">
import { ref, onMounted, nextTick } from 'vue'
import { useRouter } from 'vue-router'
import ChatBubble from '@components/business/ChatBubble.vue'
import QuickPrompts from '@components/business/QuickPrompts.vue'
import ChatInput from '@components/business/ChatInput.vue'
import { useChatStore } from '@stores/chat'
import { useAgentApi } from '@api/agent'

const router = useRouter()
const chatStore = useChatStore()
const agentApi = useAgentApi()

const messages = ref<any[]>([])
const inputMessage = ref('')
const sending = ref(false)
const chatContainer = ref<HTMLElement>()

const quickPrompts = [
  '我想查看我的客户列表和跟进状态',
  '帮我推进一下当前的服务事项和待办任务',
  '定制我的机构经纪业务经营方案'
]

onMounted(() => {
  // 加载历史消息
  messages.value = chatStore.messages
  
  // 添加欢迎消息
  if (messages.value.length === 0) {
    addMessage({
      type: 'text',
      content: '你好！我是机构经纪数字员工。我可以帮你：\n1. 查询客户信息\n2. 推进服务事项\n3. 生成经营方案\n4. 协助开户流程\n\n请问有什么可以帮你的？',
      isSelf: false,
      time: new Date()
    })
  }
})

async function sendMessage() {
  if (!inputMessage.value.trim() || sending.value) return
  
  const userMessage = inputMessage.value.trim()
  inputMessage.value = ''
  
  // 添加用户消息
  addMessage({
    type: 'text',
    content: userMessage,
    isSelf: true,
    time: new Date()
  })
  
  sending.value = true
  
  try {
    // 调用Agent接口
    const response = await agentApi.chat({
      message: userMessage,
      sessionId: chatStore.sessionId
    })
    
    // 添加机器人回复
    addMessage({
      type: response.type || 'text',
      content: response.content,
      data: response.data,
      isSelf: false,
      time: new Date()
    })
  } catch (error) {
    addMessage({
      type: 'text',
      content: '抱歉，服务暂时不可用，请稍后重试。',
      isSelf: false,
      time: new Date(),
      isError: true
    })
  } finally {
    sending.value = false
  }
}

function addMessage(msg: any) {
  messages.value.push(msg)
  chatStore.addMessage(msg)
  scrollToBottom()
}

function scrollToBottom() {
  nextTick(() => {
    if (chatContainer.value) {
      chatContainer.value.scrollTop = chatContainer.value.scrollHeight
    }
  })
}

function handleQuickPrompt(prompt: string) {
  inputMessage.value = prompt
  sendMessage()
}

function handleMessageAction(action: any) {
  // 处理消息中的操作按钮点击
  if (action.type === 'navigate') {
    router.push(action.path)
  } else if (action.type === 'api') {
    // 调用API
  }
}

function goBack() {
  router.back()
}

function handleVoiceInput() {
  // 语音输入处理
}
</script>
```

```vue
  data?: any
  isSelf: boolean
  time: Date
```

```typescript
export const useChatStore = defineStore('chat', () => {
  // State
  const sessionId = ref(localStorage.getItem('chat_session_id') || generateSessionId())
  const messages = ref<Message[]>([])
  const isConnected = ref(false)
  
  // Getters
  const lastMessage = computed(() => messages.value[messages.value.length - 1])
  const messageCount = computed(() => messages.value.length)
  
  // Actions
  function generateSessionId() {
    const id = 'session_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9)
    localStorage.setItem('chat_session_id', id)
    return id
  }
  
  function addMessage(message: Omit<Message, 'id'>) {
    const newMessage = {
      ...message,
      id: 'msg_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9)
    }
    messages.value.push(newMessage)
    
    // 只保留最近100条消息
    if (messages.value.length > 100) {
      messages.value = messages.value.slice(-100)
    }
    
    // 持久化到本地
    localStorage.setItem('chat_messages', JSON.stringify(messages.value))
  }
  
  function loadMessages() {
    const saved = localStorage.getItem('chat_messages')
    if (saved) {
      try {
        messages.value = JSON.parse(saved).map((m: any) => ({
          ...m,
          time: new Date(m.time)
        }))
      } catch (e) {
        console.error('加载历史消息失败:', e)
      }
    }
  }
  
  function clearMessages() {
    messages.value = []
    localStorage.removeItem('chat_messages')
    sessionId.value = generateSessionId()
  }
  
  function setConnected(connected: boolean) {
    isConnected.value = connected
  }
  
  return {
    sessionId,
    messages,
    isConnected,
    lastMessage,
    messageCount,
    addMessage,
    loadMessages,
    clearMessages,
    setConnected
  }
```

```typescript
import axios, { AxiosInstance, AxiosRequestConfig, AxiosError } from 'axios'
import { Toast } from 'vant'
import { useUserStore } from '@stores/user'

const request: AxiosInstance = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL || '/api',
  timeout: 30000,
  headers: {
    'Content-Type': 'application/json'
  }
})

// 请求拦截器
request.interceptors.request.use(
  (config) => {
    const userStore = useUserStore()
    if (userStore.token) {
      config.headers.Authorization = `Bearer ${userStore.token}`
    }
    return config
  },
  (error) => {
    return Promise.reject(error)
  }
)

// 响应拦截器
request.interceptors.response.use(
  (response) => {
    const { data } = response
    
    if (data.code !== 0) {
      Toast.fail(data.message || '请求失败')
      return Promise.reject(data)
    }
    
    return data.data
  },
  (error: AxiosError) => {
    if (error.response?.status === 401) {
      const userStore = useUserStore()
      userStore.clearUser()
      window.location.href = '/login'
    } else {
      Toast.fail(error.message || '网络错误')
    }
    return Promise.reject(error)
  }
)

export default request
```

```typescript
    document.documentElement.style.fontSize = `${fontSize}px`
  }
  
  setRem()
  window.addEventListener('resize', setRem)
}
```
