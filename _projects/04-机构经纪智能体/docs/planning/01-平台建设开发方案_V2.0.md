```python
class InstitutionBrokerageAgentSystem:

        self.workflow.add_node("coordinator", self.coordinator_agent)
        self.workflow.add_node("prospector", self.prospector_agent)
        self.workflow.add_node("onboarding", self.onboarding_agent)
        self.workflow.add_node("trading", self.trading_agent)
        self.workflow.add_node("service", self.service_agent)
        
        # 添加工具节点
        self.workflow.add_node("tools", self.tool_executor)
        
        # 定义边和条件路由
        self._define_edges()
        
        # 编译工作流
        self.app = self.workflow.compile()
    
    def _define_edges(self):
        """定义智能体间的协作关系"""
        
        # 协调官路由逻辑
        self.workflow.add_conditional_edges(
            "coordinator",
            self.route_by_intent,
            {
                "introduce_customer": "prospector",
                "open_account": "onboarding",
                "trading_service": "trading",
                "customer_service": "service",
                "need_tool": "tools",
                "escalate": END
            }
        )
        
        # 各智能体完成后的返回
        for agent in ["prospector", "onboarding", "trading", "service"]:
            self.workflow.add_edge(agent, "coordinator")
        
        # 工具执行后返回
        self.workflow.add_edge("tools", "coordinator")
        
        # 入口
        self.workflow.set_entry_point("coordinator")


class CoordinatorAgent:
    """机构经纪协调官 - 中央调度智能体"""
    
    def __init__(self, llm, tools, memory):
        self.llm = llm
        self.tools = tools
        self.memory = memory
        
        # 系统提示词（工程级）
        self.system_prompt = """
        你是机构经纪业务协调官，负责统筹四大业务目标的执行。
        
        核心职责：
        1. 意图识别：识别用户请求属于【引入客户】【账户开立】【交易服务】【客户服务】哪一类
        2. 任务路由：将任务分配给对应的超级数字员工
        3. 进度跟踪：监控任务执行状态，确保闭环
        4. 异常升级：识别需公司层面决策的事项，启动请示流程
        
        决策规则：
        - 战略级客户特殊政策 → 触发【向上请示】
        - 跨部门资源协调 → 触发【横向协同】
        - 合规风险事件 → 触发【紧急请示】
        - 常规业务 → 分配给对应Agent执行
        
        底座调用：
        - 统一语料：查询客户信息、舆情监控
        - 关系网络：关联方分析、风险传导
        - 风控规则：准入审核、限额校验
        """
    
    async def invoke(self, state: AgentState) -> AgentState:
        """协调官执行逻辑"""
        
        # 1. 加载历史上下文
        context = await self.memory.load_context(state["session_id"])
        
        # 2. 意图识别
        intent = await self._classify_intent(state["input"], context)
        
        # 3. 判断是否需要请示
        if self._need_escalation(intent, state):
            return {**state, "next": "escalate", "escalation_reason": self._get_reason()}
        
        # 4. 判断是否需要工具
        if self._need_tool(intent):
            return {**state, "next": "tools", "tool_request": self._prepare_tool_request()}
        
        # 5. 路由到对应Agent
        target_agent = self._route_to_agent(intent)
        return {**state, "next": target_agent, "intent": intent}
    
    async def _classify_intent(self, input_text: str, context: dict) -> Intent:
        """意图分类 - 使用本地分类模型"""
        
        # 调用本地意图分类模型（非底座，延迟更低）
        response = await self.llm.acomplete(
            model="local/intent-classifier-v2",
            messages=[
                {"role": "system", "content": "将用户输入分类到以下意图之一..."},
                {"role": "user", "content": input_text}
            ],
            temperature=0.1
        )
        
        return Intent.parse(response.content)


class ProspectorAgent:
    """客户开发员Agent - 引入客户专家"""
    
    def __init__(self, llm, corpus_adapter, risk_adapter):
        self.llm = llm
        self.corpus = corpus_adapter
        self.risk = risk_adapter
        
        self.system_prompt = """
        你是客户开发专家，负责线索挖掘、资质评估、客户画像。
        
        工作流程：
        1. 线索收集：接收来自AI挖掘、客户主动咨询、主动 outreach 的线索
        2. 资质初筛：调用底座风控规则，评估准入条件
        3. 客户画像：整合工商、财务、舆情数据，生成全景画像
        4. 分配跟进：AI评估后分配给适格客户经理
        
        工具调用：
        - search_institution: 查询机构工商信息
        - evaluate_qualification: 资质评估
        - generate_profile: 生成客户画像
        - assign_manager: 分配客户经理
        """
    
    async def invoke(self, state: AgentState) -> AgentState:
        """客户开发执行逻辑"""
        
        # 获取任务参数
        task = state.get("task", {})
        lead_source = task.get("source")  # ai_mining / inbound / outbound
        
        if lead_source == "ai_mining":
            return await self._process_ai_leads(state)
        elif lead_source == "inbound":
            return await self._process_inbound(state)
        elif lead_source == "outbound":
            return await self._process_outbound(state)
        else:
            return await self._general_prospecting(state)
    
    async def _process_ai_leads(self, state: AgentState) -> AgentState:
        """处理AI挖掘的线索"""
        
        # 1. 调用底座统一语料获取机构信息
        institution_info = await self.corpus.query_institution(
            name=state["input"],
            fields=["basic", "financial", "news", "risk"]
        )
        
        # 2. 调用底座风控进行资质评估
        qualification = await self.risk.evaluate_qualification(
            institution_id=institution_info["id"],
            business_type="institution_brokerage"
        )
        
        # 3. AI评估生成报告
        assessment = await self.llm.acomplete(
            model="local/qwen-72b",
            messages=[
                {"role": "system", "content": self.system_prompt},
                {"role": "user", "content": f"请评估以下客户：\n{institution_info}\n{qualification}"}
            ],
            tools=["generate_score", "recommend_priority"]
        )
        
        # 4. 生成客户画像并分配
        profile = await self._generate_profile(institution_info, qualification, assessment)
        assignment = await self._assign_manager(profile)
        
        return {
            **state,
            "output": {
                "profile": profile,
                "assignment": assignment,
                "next_action": "客户经理跟进"
            }
        }
```

```python
# Agent-to-Agent 通信协议
class A2AProtocol:
    """智能体间通信标准协议"""
    
    MESSAGE_SCHEMA = {
        "message_id": "uuid",
        "sender": "agent_id",
        "receiver": "agent_id",
        "message_type": "task_request|task_response|escalation|notification",
        "payload": {
            "task_id": "uuid",
            "intent": "string",
            "context": "dict",
            "data": "any",
            "expected_output": "dict"
        },
        "priority": "high|medium|low",
        "timestamp": "iso8601",
        "signature": "hmac_sha256"
    }
    
    async def send_task(self, sender: str, receiver: str, task: dict) -> TaskReceipt:
        """发送任务到目标智能体"""
        
        message = {
            "message_id": str(uuid4()),
            "sender": sender,
            "receiver": receiver,
            "message_type": "task_request",
            "payload": task,
            "priority": self._calculate_priority(task),
            "timestamp": datetime.now().isoformat()
        }
        
        # 签名
        message["signature"] = self._sign(message)
        
        # 发送到消息队列
        await self.message_queue.publish(
            exchange="a2a.exchange",
            routing_key=f"agent.{receiver}",
            message=json.dumps(message)
        )
        
        return TaskReceipt(message_id=message["message_id"])
    
    async def receive_task(self, agent_id: str) -> AsyncIterator[dict]:
        """接收发给当前智能体的任务"""
        
        queue = await self.message_queue.declare_queue(
            f"agent.{agent_id}.queue",
            durable=True
        )
        
        async for message in queue.consume():
            # 验证签名
            if not self._verify_signature(message):
                logger.warning(f"Invalid signature from {message['sender']}")
                continue
            
            yield message
```

### 2.3 记忆系统设计

```python
# 分层记忆系统
class HierarchicalMemory:
    """三级记忆系统：会话记忆、工作记忆、长期记忆"""
    
    def __init__(self):
        # 会话级缓存 (Redis)
        self.session_cache = RedisCache(ttl=3600)
        
        # 工作记忆 (PostgreSQL + pgvector)
        self.working_memory = WorkingMemoryStore()
        
        # 长期记忆 (Milvus向量库 + 图数据库)
        self.long_term_memory = LongTermMemoryStore()
    
    async def save(self, session_id: str, agent_id: str, memory_type: str, content: dict):
        """保存记忆"""
        
        if memory_type == "session":
            # 短期会话记忆
            await self.session_cache.set(
                f"{session_id}:{agent_id}",
                content,
                ttl=3600
            )
            
        elif memory_type == "working":
            # 工作任务记忆
            embedding = await self._embed(content)
            await self.working_memory.upsert(
                id=f"{session_id}:{agent_id}:{content['task_id']}",
                vector=embedding,
                metadata=content
            )
            
        elif memory_type == "long_term":
            # 长期业务知识
            embedding = await self._embed(content)
            
            # 存入向量库
            await self.long_term_memory.vector_store.insert(
                vectors=[embedding],
                entities=[{
                    "agent_id": agent_id,
                    "content": content,
                    "timestamp": datetime.now()
                }]
            )
            
            # 构建关系图谱
            if "related_entities" in content:
                await self.long_term_memory.graph_store.add_relations(
                    source=content["entity_id"],
                    relations=content["related_entities"]
                )
    
    async def retrieve(self, session_id: str, agent_id: str, query: str, k: int = 5) -> list:
        """检索相关记忆"""
        
        # 1. 检索会话上下文
        session_context = await self.session_cache.get(f"{session_id}:{agent_id}")
        
        # 2. 向量检索工作记忆
        query_embedding = await self._embed(query)
        working_memories = await self.working_memory.search(
            vector=query_embedding,
            filter={"agent_id": agent_id},
            top_k=k
        )
        
        # 3. 检索长期记忆（混合检索：向量+图）
        long_term_vectors = await self.long_term_memory.vector_store.search(
            vector=query_embedding,
            top_k=k
        )
        
        # 图遍历获取关联信息
        graph_context = await self.long_term_memory.graph_store.traverse(
            start_nodes=[m["entity_id"] for m in long_term_vectors],
            depth=2
        )
        
        return {
            "session": session_context,
            "working": working_memories,
            "long_term": long_term_vectors,
            "graph_context": graph_context
        }
```

```python
                )
                
                gpu_util = metrics["value"]
                pending_requests = await self.metrics.query(
                    f'vllm:num_requests_waiting{{model="{model_name}"}}'
                )
                
                # 扩容策略
                if gpu_util > 0.8 or pending_requests > 20:
                    await self._scale_up(model_name, replicas=+1)
                    
                # 缩容策略
                elif gpu_util < 0.3 and pending_requests < 5:
                    current_replicas = await self._get_replicas(model_name)
                    if current_replicas > 1:
                        await self._scale_down(model_name, replicas=-1)
    
    async def smart_routing(self, request: dict) -> str:
        """智能路由到最优实例"""
        
        model_name = request["model"]
        
        # 获取所有健康实例
        instances = await self._get_healthy_instances(model_name)
        
        # 评分选择最优实例
        best_instance = None
        best_score = float('-inf')
        
        for instance in instances:
            score = self._calculate_instance_score(instance, request)
            if score > best_score:
                best_score = score
                best_instance = instance
        
        return best_instance["endpoint"]
    
    def _calculate_instance_score(self, instance: dict, request: dict) -> float:
        """计算实例评分"""
        
        # 负载评分（越低越好）
        load_score = (1 - instance["gpu_utilization"]) * 40
        
        # 队列深度评分（越低越好）
        queue_score = (1 - instance["queue_depth"] / 100) * 30
        
        # KV Cache命中率评分（越高越好）
        cache_score = instance["kv_cache_hit_rate"] * 20
        
        # 网络延迟评分（越低越好）
        latency_score = (1 - instance["avg_latency_ms"] / 1000) * 10
        
        return load_score + queue_score + cache_score + latency_score
```

```python
# 语料融合引擎
class CorpusFusionEngine:
    """多源语料融合与冲突消解"""
    
    def __init__(self):
        self.entity_aligner = EntityAligner()
        self.conflict_resolver = ConflictResolver()
        self.knowledge_graph = KnowledgeGraphBuilder()
    
    async def fuse(
        self, 
        local_results: list, 
        base_results: list,
        internet_results: list
    ) -> FusedResult:
        """融合多源语料"""
        
        # 1. 实体对齐
        aligned = await self.entity_aligner.align(
            local_results + base_results + internet_results
        )
        
        # 2. 冲突检测
        conflicts = self._detect_conflicts(aligned)
        
        # 3. 冲突消解
        resolved = await self.conflict_resolver.resolve(conflicts)
        
        # 4. 可信度评分
        scored = self._calculate_confidence(resolved)
        
        # 5. 构建融合结果
        return FusedResult(
            entities=scored,
            sources={
                "local": len(local_results),
                "base": len(base_results),
                "internet": len(internet_results)
            },
            confidence_distribution=self._analyze_confidence(scored)
        )
    
    def _detect_conflicts(self, aligned: list) -> list[Conflict]:
        """检测语料冲突"""
        
        conflicts = []
        
        for entity_id, mentions in groupby(aligned, key=lambda x: x["entity_id"]):
            mentions = list(mentions)
            
            # 属性值冲突检测
            for attr in ["registered_capital", "legal_representative"]:
                values = set(m[attr] for m in mentions if attr in m)
                if len(values) > 1:
                    conflicts.append(Conflict(
                        type="attribute_mismatch",
                        entity_id=entity_id,
                        attribute=attr,
                        values=values,
                        sources=[m["source"] for m in mentions]
                    ))
            
            # 时间线冲突检测
            dates = [m["update_time"] for m in mentions if "update_time" in m]
            if dates and max(dates) - min(dates) > timedelta(days=30):
                conflicts.append(Conflict(
                    type="temporal_inconsistency",
                    entity_id=entity_id,
                    time_range=(min(dates), max(dates)),
                    sources=[m["source"] for m in mentions]
                ))
        
        return conflicts
    
    async def _resolve_conflict(self, conflict: Conflict) -> ResolvedFact:
        """冲突消解策略"""
        
        if conflict.type == "attribute_mismatch":
            # 策略1：按数据源可信度排序
            source_priority = ["local", "base", "internet"]
            
            for source in source_priority:
                for mention in conflict.mentions:
                    if mention["source"] == source:
                        return ResolvedFact(
                            value=mention[conflict.attribute],
                            source=source,
                            confidence=mention.get("confidence", 0.8),
                            reason=f"优先使用{source}数据源"
                        )
            
        elif conflict.type == "temporal_inconsistency":
            # 策略2：取最新数据
            latest = max(conflict.mentions, key=lambda x: x["update_time"])
            return ResolvedFact(
                value=latest[conflict.attribute],
                source=latest["source"],
                confidence=0.9,
                reason="使用最新更新时间的数据"
            )
```

```python
            timeout=30.0,
            http2=True
        )
        
        # 本地缓存
        self.cache = RedisCache(ttl=300)
        
        # 断路器
        self.circuit_breaker = CircuitBreaker(
            failure_threshold=5,
            recovery_timeout=60
        )
    
    async def _get_access_token(self) -> str:
        """获取访问令牌"""
        
        cache_key = "base_access_token"
        token = await self.cache.get(cache_key)
        
        if not token:
            token_data = await self.auth_client.fetch_token()
            token = token_data["access_token"]
            expires_in = token_data.get("expires_in", 3600)
            await self.cache.set(cache_key, token, ttl=expires_in - 60)
        
        return token
    
    async def _call_base_api(
        self, 
        endpoint: str, 
        payload: dict,
        use_cache: bool = False
    ) -> dict:
        """调用底座API"""
        
        # 缓存检查
        cache_key = f"base:{endpoint}:{hash(str(payload))}"
        if use_cache:
            cached = await self.cache.get(cache_key)
            if cached:
                return cached
        
        # 断路器保护
        if not self.circuit_breaker.allow_request():
            raise BaseModelUnavailable("底座服务熔断")
        
        try:
            # 获取令牌
            token = await self._get_access_token()
            
            # 请求签名
            timestamp = str(int(time.time()))
            nonce = str(uuid4())
            sign = self._sign_request(payload, timestamp, nonce)
            
            # 发送请求
            response = await self.http_client.post(
                endpoint,
                json=payload,
                headers={
                    "Authorization": f"Bearer {token}",
                    "X-Request-Time": timestamp,
                    "X-Request-Nonce": nonce,
                    "X-Request-Sign": sign,
                    "X-Service-ID": self.config.SERVICE_ID,
                    "X-Trace-ID": str(uuid4())
                }
            )
            
            response.raise_for_status()
            result = response.json()
            
            # 记录成功
            self.circuit_breaker.record_success()
            
            # 缓存结果
            if use_cache:
                await self.cache.set(cache_key, result, ttl=300)
            
            return result
            
        except httpx.HTTPStatusError as e:
            self.circuit_breaker.record_failure()
            raise BaseModelError(f"底座API错误: {e.response.status_code}")
            
        except Exception as e:
            self.circuit_breaker.record_failure()
            raise BaseModelError(f"底座调用失败: {str(e)}")
    
    def _sign_request(self, payload: dict, timestamp: str, nonce: str) -> str:
        """请求签名（HMAC-SHA256）"""
        
        sign_str = f"{self.config.CLIENT_ID}:{timestamp}:{nonce}:{json.dumps(payload, sort_keys=True)}"
        return hmac.new(
            self.config.SIGN_KEY.encode(),
            sign_str.encode(),
            hashlib.sha256
        ).hexdigest()
```

```typescript
// 核心SDK
class CoreSDK {
  private connection: ConnectionManager;
  private messageQueue: MessageQueue;
  private stateManager: StateManager;
  private eventBus: EventEmitter;
  
  constructor(config: ClientConfig) {
    this.connection = new ConnectionManager(config);
    this.messageQueue = new MessageQueue();
    this.stateManager = new StateManager();
    this.eventBus = new EventEmitter();
    
    this.initialize();
  }
  
  private async initialize(): Promise<void> {
    // 建立与服务端的连接
    await this.connection.connect();
    
    // 注册消息处理器
    this.connection.onMessage(this.handleServerMessage.bind(this));
    
    // 启动消息队列处理器
    this.startMessageProcessor();
    
    // 启动状态同步
    this.startStateSync();
  }
  
  async send(message: ClientMessage): Promise<void> {
    // 添加到消息队列
    await this.messageQueue.enqueue(message);
    
    // 本地乐观更新
    this.stateManager.optimisticUpdate(message);
    
    // 触发本地事件
    this.eventBus.emit('message-sent', message);
  }
  
  async createConversation(options: ConversationOptions): Promise<Conversation> {
    const response = await this.connection.request({
      type: 'create_conversation',
      payload: options
    });
    
    const conversation: Conversation = {
      id: response.conversation_id,
      agentType: options.agentType,
      status: 'active',
      createdAt: Date.now()
    };
    
    // 保存会话状态
    this.stateManager.setConversation(conversation);
    
    return conversation;
  }
  
  async processMessage(message: Message): Promise<AgentResponse> {
    // 发送到服务端处理
    const response = await this.connection.request({
      type: 'process_message',
      payload: {
        message,
        conversation_id: this.stateManager.getCurrentConversation()?.id,
        context: this.stateManager.getContext()
      }
    });
    
    return response;
  }
  
  private async handleServerMessage(message: ServerMessage): Promise<void> {
    switch (message.type) {
      case 'agent_response':
        // 更新状态
        this.stateManager.applyUpdate(message.payload);
        
        // 触发事件
        this.eventBus.emit('agent-response', message.payload);
        break;
        
      case 'state_sync':
        // 状态同步
        this.stateManager.sync(message.payload);
        break;
        
      case 'notification':
        // 通知消息
        this.eventBus.emit('notification', message.payload);
        break;
        
      case 'typing_start':
        this.eventBus.emit('typing-start', message.payload);
        break;
        
      case 'typing_end':
        this.eventBus.emit('typing-end', message.payload);
        break;
    }
  }
  
  private startMessageProcessor(): void {
    // 消费消息队列
    this.messageQueue.consume(async (message) => {
      try {
        await this.connection.send(message);
        this.messageQueue.ack(message.id);
      } catch (error) {
        // 重试逻辑
        if (message.retryCount < 3) {
          message.retryCount++;
          await this.messageQueue.requeue(message);
        } else {
          this.messageQueue.dlq(message);
          this.eventBus.emit('message-failed', message);
        }
      }
    });
  }
  
  private startStateSync(): void {
    // 定期状态同步
    setInterval(async () => {
      const syncRequest = this.stateManager.getSyncRequest();
      const syncResponse = await this.connection.request({
        type: 'state_sync',
        payload: syncRequest
      });
      this.stateManager.applySync(syncResponse);
    }, 5000);
  }
  
  // 事件订阅API
  on(event: string, callback: Function): void {
    this.eventBus.on(event, callback);
  }
  
  off(event: string, callback: Function): void {
    this.eventBus.off(event, callback);
  }
}

// 连接管理器
class ConnectionManager {
  private ws: WebSocket | null = null;
  private http: AxiosInstance;
  private reconnectAttempts = 0;
  private maxReconnectAttempts = 5;
  
  constructor(private config: ClientConfig) {
    this.http = axios.create({
      baseURL: config.apiEndpoint,
      timeout: 30000
    });
  }
  
  async connect(): Promise<void> {
    return new Promise((resolve, reject) => {
      this.ws = new WebSocket(this.config.wsEndpoint);
      
      this.ws.onopen = () => {
        this.reconnectAttempts = 0;
        
        // 发送认证消息
        this.ws?.send(JSON.stringify({
          type: 'auth',
          token: this.config.authToken
        }));
        
        resolve();
      };
      
      this.ws.onerror = (error) => {
        reject(error);
      };
      
      this.ws.onclose = () => {
        this.handleDisconnect();
      };
    });
  }
  
  private handleDisconnect(): void {
    if (this.reconnectAttempts < this.maxReconnectAttempts) {
      this.reconnectAttempts++;
      const delay = Math.min(1000 * Math.pow(2, this.reconnectAttempts), 30000);
      
      setTimeout(() => {
        this.connect();
      }, delay);
    }
  }
  
  async send(message: any): Promise<void> {
    if (this.ws?.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify(message));
    } else {
      // WebSocket不可用，降级到HTTP
      await this.http.post('/messages', message);
    }
  }
  
  async request(payload: any): Promise<any> {
    const response = await this.http.post('/request', payload);
    return response.data;
  }
  
  onMessage(callback: (message: any) => void): void {
    this.ws?.addEventListener('message', (event) => {
      const message = JSON.parse(event.data);
      callback(message);
    });
  }
}
```

### 6.5 客户端状态管理

```typescript
// 状态管理器（支持离线优先）
class StateManager {
  private db: IDBDatabase;
  private memoryState: AppState;
  private syncQueue: Array<StateChange> = [];
  
  async initialize(): Promise<void> {
    // 初始化IndexedDB
    this.db = await openDB('InstitutionBrokerage', 1, {
      upgrade(db) {
        db.createObjectStore('conversations', { keyPath: 'id' });
        db.createObjectStore('messages', { keyPath: 'id' });
        db.createObjectStore('syncQueue', { keyPath: 'id', autoIncrement: true });
      }
    });
    
    // 从本地存储恢复状态
    await this.restoreFromStorage();
  }
  
  async restoreFromStorage(): Promise<void> {
    const conversations = await this.db.getAll('conversations');
    const messages = await this.db.getAll('messages');
    
    this.memoryState = {
      conversations: new Map(conversations.map(c => [c.id, c])),
      messages: new Map(messages.map(m => [m.id, m])),
      currentConversationId: null,
      unreadCount: 0
    };
  }
  
  optimisticUpdate(change: StateChange): void {
    // 乐观更新内存状态
    this.applyChangeToMemory(change);
    
    // 添加到同步队列
    this.syncQueue.push(change);
    
    // 异步持久化
    this.persistChange(change);
  }
  
  async applyServerUpdate(update: ServerUpdate): Promise<void> {
    // 检查是否与本地乐观更新冲突
    const conflicts = this.detectConflicts(update);
    
    if (conflicts.length > 0) {
      // 解决冲突
      const resolved = this.resolveConflicts(conflicts, update);
      await this.applyResolvedUpdate(resolved);
    } else {
      await this.applyChangeToMemoryAndStorage(update);
    }
    
    // 通知UI更新
    this.emit('state-changed', update);
  }
  
  private async persistChange(change: StateChange): Promise<void> {
    const tx = this.db.transaction(['syncQueue', 'conversations', 'messages'], 'readwrite');
    
    // 添加到同步队列
    await tx.objectStore('syncQueue').add(change);
    
    // 更新对应存储
    if (change.type === 'conversation') {
      await tx.objectStore('conversations').put(change.data);
    } else if (change.type === 'message') {
      await tx.objectStore('messages').put(change.data);
    }
    
    await tx.done;
  }
}
```
