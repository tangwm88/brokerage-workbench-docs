# 本地语料管理系统
class LocalCorpusManager:
    """本地语料采集、处理、存储"""
    
    def __init__(self):
        # 数据采集器
        self.collectors = {
            "crm": CRMCollector(),          # CRM系统
            "oa": OACollector(),            # OA系统
            "trading": TradingCollector(),  # 交易系统
            "research": ResearchCollector() # 研报系统
        }
        
        # 数据处理管道
        self.pipeline = DataPipeline([
            PIIAnonymizer(),        # 脱敏
            EntityExtractor(),      # 实体抽取
            RelationExtractor(),    # 关系抽取
            Vectorizer()            # 向量化
        ])
        
        # 存储层
        self.vector_store = MilvusClient()
        self.graph_store = Neo4jClient()
        self.doc_store = MongoClient()
    
    async def ingest_document(self, source: str, doc: dict):
        """文档入库流程"""
        
        # 1. 数据采集
        raw_data = await self.collectors[source].fetch(doc["id"])
        
        # 2. 数据清洗与脱敏
        clean_data = await self._clean_and_anonymize(raw_data)
        
        # 3. 实体抽取
        entities = await self._extract_entities(clean_data["content"])
        
        # 4. 向量化
        embedding = await self._embed(clean_data["content"])
        
        # 5. 多路存储
        await asyncio.gather(
            # 存入向量库（语义检索）
            self.vector_store.insert(
                collection="local_corpus",
                vectors=[embedding],
                entities=[{
                    "doc_id": doc["id"],
                    "source": source,
                    "content": clean_data["content"],
                    "metadata": clean_data["metadata"],
                    "timestamp": datetime.now()
                }]
            ),
            
            # 存入图数据库（关系分析）
            self.graph_store.add_document(
                doc_id=doc["id"],
                entities=entities,
                relations=entities.get("relations", [])
            ),
            
            # 存入文档库（原始存储）
            self.doc_store.insert_one("documents", clean_data)
        )
    
    async def semantic_search(
        self, 
        query: str, 
        filters: dict = None,
        top_k: int = 10
    ) -> list[SearchResult]:
        """语义检索"""
        
        # 查询向量化
        query_embedding = await self._embed(query)
        
        # 混合检索（向量 + 关键词）
        vector_results = await self.vector_store.search(
            collection="local_corpus",
            vector=query_embedding,
            filter=filters,
            top_k=top_k * 2  # 扩大召回
        )
        
        # 重排序（Cross-Encoder）
        reranked = await self._rerank(query, vector_results, top_k)
        
        return reranked

class InternetCorpusCollector:
    """外部公开信息采集"""
    
    def __init__(self):
        self.sources = {
            "news": NewsCollector(),           # 财经新闻
            "announcement": AnnouncementCollector(),  # 公告
            "social": SocialMediaCollector(),  # 社交媒体
            "industry": IndustryReportCollector()     # 行业报告
        }
        
        # 内容去重器
        self.deduplicator = SimHashDeduplicator()
        
        # 质量评估器
        self.quality_checker = ContentQualityChecker()
    
    async def monitor_institution(
        self, 
        institution_name: str,
        alert_threshold: float = 0.7
    ) -> AsyncIterator[CorpusEvent]:
        """机构舆情实时监控"""
        
        # 启动多源监控
        tasks = [
            self.sources["news"].monitor(institution_name),
            self.sources["social"].monitor(institution_name),
            self.sources["announcement"].monitor(institution_name)
        ]
        
        # 合并流
        merged = merge_async_iterators(*tasks)
        
        async for event in merged:
            # 去重检查
            if await self.deduplicator.is_duplicate(event["content"]):
                continue
            
            # 情感分析
            sentiment = await self._analyze_sentiment(event["content"])
            
            # 质量评分
            quality_score = await self.quality_checker.score(event)
            
            # 风险判断
            risk_level = self._calculate_risk(sentiment, quality_score)
            
            if risk_level >= alert_threshold:
                yield CorpusEvent(
                    type="risk_alert",
                    institution=institution_name,
                    content=event["content"],
                    sentiment=sentiment,
                    risk_level=risk_level,
                    source=event["source"],
                    timestamp=datetime.now()
                )

```python
# 语料融合引擎
class CorpusFusionEngine:
    """多源语料融合与冲突消解"""
    
    def __init__(self):
        self.entity_aligner = EntityAligner()
        self.conflict_resolver = ConflictResolver()
        self.knowledge_graph = KnowledgeGraphBuilder()
    
    async def fuse(
        self, 
        local_results: list, 
        base_results: list,
        internet_results: list
    ) -> FusedResult:
        """融合多源语料"""
        
        # 1. 实体对齐
        aligned = await self.entity_aligner.align(
            local_results + base_results + internet_results
        )
        
        # 2. 冲突检测
        conflicts = self._detect_conflicts(aligned)
        
        # 3. 冲突消解
        resolved = await self.conflict_resolver.resolve(conflicts)
        
        # 4. 可信度评分
        scored = self._calculate_confidence(resolved)
        
        # 5. 构建融合结果
        return FusedResult(
            entities=scored,
            sources={
                "local": len(local_results),
                "base": len(base_results),
                "internet": len(internet_results)
            },
            confidence_distribution=self._analyze_confidence(scored)
        )
    
    def _detect_conflicts(self, aligned: list) -> list[Conflict]:
        """检测语料冲突"""
        
        conflicts = []
        
        for entity_id, mentions in groupby(aligned, key=lambda x: x["entity_id"]):
            mentions = list(mentions)
            
            # 属性值冲突检测
            for attr in ["registered_capital", "legal_representative"]:
                values = set(m[attr] for m in mentions if attr in m)
                if len(values) > 1:
                    conflicts.append(Conflict(
                        type="attribute_mismatch",
                        entity_id=entity_id,
                        attribute=attr,
                        values=values,
                        sources=[m["source"] for m in mentions]
                    ))
            
            # 时间线冲突检测
            dates = [m["update_time"] for m in mentions if "update_time" in m]
            if dates and max(dates) - min(dates) > timedelta(days=30):
                conflicts.append(Conflict(
                    type="temporal_inconsistency",
                    entity_id=entity_id,
                    time_range=(min(dates), max(dates)),
                    sources=[m["source"] for m in mentions]
                ))
        
        return conflicts
    
    async def _resolve_conflict(self, conflict: Conflict) -> ResolvedFact:
        """冲突消解策略"""
        
        if conflict.type == "attribute_mismatch":
            # 策略1：按数据源可信度排序
            source_priority = ["local", "base", "internet"]
            
            for source in source_priority:
                for mention in conflict.mentions:
                    if mention["source"] == source:
                        return ResolvedFact(
                            value=mention[conflict.attribute],
                            source=source,
                            confidence=mention.get("confidence", 0.8),
                            reason=f"优先使用{source}数据源"
                        )
            
        elif conflict.type == "temporal_inconsistency":
            # 策略2：取最新数据
            latest = max(conflict.mentions, key=lambda x: x["update_time"])
            return ResolvedFact(
                value=latest[conflict.attribute],
                source=latest["source"],
                confidence=0.9,
                reason="使用最新更新时间的数据"
            )
```

# 底座统一SDK
class BaseModelSDK:
    """公司自动驾驶大模型底座统一SDK"""
    
    def __init__(self, config: BaseConfig):
        self.config = config
        
        # 认证客户端
        self.auth_client = OAuth2Client(
            token_endpoint=config.TOKEN_URL,
            client_id=config.CLIENT_ID,
            client_secret=config.CLIENT_SECRET
        )
        
        # HTTP客户端（mTLS）
        self.http_client = httpx.AsyncClient(
            base_url=config.BASE_API_URL,
            cert=(config.CLIENT_CERT, config.CLIENT_KEY),
            verify=config.CA_CERT,
            timeout=30.0,
            http2=True
        )
        
        # 本地缓存
        self.cache = RedisCache(ttl=300)
        
        # 断路器
        self.circuit_breaker = CircuitBreaker(
            failure_threshold=5,
            recovery_timeout=60
        )
    
    async def _get_access_token(self) -> str:
        """获取访问令牌"""
        
        cache_key = "base_access_token"
        token = await self.cache.get(cache_key)
        
        if not token:
            token_data = await self.auth_client.fetch_token()
            token = token_data["access_token"]
            expires_in = token_data.get("expires_in", 3600)
            await self.cache.set(cache_key, token, ttl=expires_in - 60)
        
        return token
    
    async def _call_base_api(
        self, 
        endpoint: str, 
        payload: dict,
        use_cache: bool = False
    ) -> dict:
        """调用底座API"""
        
        # 缓存检查
        cache_key = f"base:{endpoint}:{hash(str(payload))}"
        if use_cache:
            cached = await self.cache.get(cache_key)
            if cached:
                return cached
        
        # 断路器保护
        if not self.circuit_breaker.allow_request():
            raise BaseModelUnavailable("底座服务熔断")
        
        try:
            # 获取令牌
            token = await self._get_access_token()
            
            # 请求签名
            timestamp = str(int(time.time()))
            nonce = str(uuid4())
            sign = self._sign_request(payload, timestamp, nonce)
            
            # 发送请求
            response = await self.http_client.post(
                endpoint,
                json=payload,
                headers={
                    "Authorization": f"Bearer {token}",
                    "X-Request-Time": timestamp,
                    "X-Request-Nonce": nonce,
                    "X-Request-Sign": sign,
                    "X-Service-ID": self.config.SERVICE_ID,
                    "X-Trace-ID": str(uuid4())
                }
            )
            
            response.raise_for_status()
            result = response.json()
            
            # 记录成功
            self.circuit_breaker.record_success()
            
            # 缓存结果
            if use_cache:
                await self.cache.set(cache_key, result, ttl=300)
            
            return result
            
        except httpx.HTTPStatusError as e:
            self.circuit_breaker.record_failure()
            raise BaseModelError(f"底座API错误: {e.response.status_code}")
            
        except Exception as e:
            self.circuit_breaker.record_failure()
            raise BaseModelError(f"底座调用失败: {str(e)}")
    
    def _sign_request(self, payload: dict, timestamp: str, nonce: str) -> str:
        """请求签名（HMAC-SHA256）"""
        
        sign_str = f"{self.config.CLIENT_ID}:{timestamp}:{nonce}:{json.dumps(payload, sort_keys=True)}"
        return hmac.new(
            self.config.SIGN_KEY.encode(),
            sign_str.encode(),
            hashlib.sha256
        ).hexdigest()

class BaseDataFeedbackManager:

    ):

        feedback = {
            "type": "customer_interaction",
            "timestamp": datetime.now().isoformat(),
            "data": {
                "session_id": session_id,
                "customer_id": customer_id,
                "intent": interaction["intent"],
                "satisfaction": interaction.get("satisfaction"),
                "resolution": interaction.get("resolution"),
                "feedback_text": interaction.get("feedback")
            }
        }
        
        await self.feedback_queue.send(feedback)
    
    async def feedback_business_outcome(
        self,
        task_id: str,
        task_type: str,
        outcome: dict
    ):
        """回流业务成果数据"""
        
        feedback = {
            "type": "business_outcome",
            "timestamp": datetime.now().isoformat(),
            "data": {
                "task_id": task_id,
                "task_type": task_type,
                "success": outcome["success"],
                "revenue": outcome.get("revenue"),
                "duration": outcome["duration"],
                "ai_contribution": outcome.get("ai_contribution", 0)
            }
        }
        
        await self.feedback_queue.send(feedback)
