## 一、实施目标

### 1.1 核心目标
- **数据不出域**：所有业务数据、客户数据、交互记录存储于公司内网
- **底座可对接**：与公司级自动驾驶大模型底座无缝集成
- **渐进式落地**：支持从试点到全面推广的分阶段实施
- **合规可审计**：满足证券行业监管要求，完整操作日志

### 1.2 实施范围
| 模块 | 本地化要求 | 优先级 |
|------|-----------|--------|
| 智能体服务层 | 完全内网部署 | P0 |
| 记忆系统 | 内网数据库 | P0 |
| 底座适配层 | 内网API网关 | P0 |
| 客户端 | 飞书/企业微信集成 | P1 |
| 监控系统 | 内网Prometheus | P1 |

### 4.2 底座接口封装

```python
# 底座适配器基类
class BaseAdapter:
    """底座能力适配器基类"""
    
    def __init__(self):
        self.client = httpx.AsyncClient(
            base_url=settings.BASE_API_URL,
            cert=(settings.CLIENT_CERT, settings.CLIENT_KEY),
            verify=settings.CA_CERT,
            timeout=30.0
        )
    
    async def call_base(self, endpoint: str, payload: dict) -> dict:
        """调用底座接口"""
        # 添加请求签名
        signature = self._sign_request(payload)
        headers = {
            "X-Service-Id": settings.SERVICE_ID,
            "X-Request-Sign": signature,
            "X-Timestamp": str(int(time.time()))
        }
        
        response = await self.client.post(
            endpoint,
            json=payload,
            headers=headers
        )
        response.raise_for_status()
        return response.json()

# 统一语料适配器
class UnifiedCorpusAdapter(BaseAdapter):
    """统一语料底座适配"""
    
    async def query_institution_info(self, institution_name: str) -> dict:
        """查询机构信息"""
        return await self.call_base("/corpus/institution/query", {
            "name": institution_name,
            "fields": ["basic", "financial", "risk", "news"],
            "cache_ttl": 3600
        })
    
    async def monitor_sentiment(self, institution_ids: list) -> dict:
        """监控舆情"""
        return await self.call_base("/corpus/sentiment/monitor", {
            "targets": institution_ids,
            "channels": ["news", "weibo", "forum"],
            "alert_threshold": 0.7
        })

# 关系网络适配器
class RelationshipNetworkAdapter(BaseAdapter):
    """关系网络底座适配"""
    
    async def query_related_parties(self, institution_id: str) -> dict:
        """查询关联方"""
        return await self.call_base("/network/related/query", {
            "institution_id": institution_id,
            "depth": 2,
            "relation_types": ["shareholder", "investment", "guarantee"]
        })
    
    async def detect_risk_chain(self, institution_id: str) -> dict:
        """风险链检测"""
        return await self.call_base("/network/risk/chain", {
            "institution_id": institution_id,
            "risk_types": ["credit", "market", "operational"]
        })

# 风控规则适配器
class RiskControlAdapter(BaseAdapter):
    """风控规则底座适配"""
    
    async def check_account_opening(self, application: dict) -> dict:
        """开户准入检查"""
        return await self.call_base("/risk/account/check", {
            "application": application,
            "check_items": ["compliance", "credit", "aml"]
        })
    
    async def get_trading_limits(self, account_id: str) -> dict:
        """获取交易限额"""
        return await self.call_base("/risk/trading/limits", {
            "account_id": account_id
        })
```
