### 2.2 表结构详细定义

#### 2.2.1 users (用户表)

```sql
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    employee_id VARCHAR(50) UNIQUE NOT NULL COMMENT '员工编号',
    name VARCHAR(100) NOT NULL COMMENT '姓名',
    department VARCHAR(100) COMMENT '部门',
    role VARCHAR(50) NOT NULL COMMENT '角色: manager/sales/admin',
    permissions JSONB DEFAULT '{}' COMMENT '权限配置',
    status VARCHAR(20) DEFAULT 'active' COMMENT '状态: active/inactive',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 索引
CREATE INDEX idx_users_employee_id ON users(employee_id);
CREATE INDEX idx_users_department ON users(department);
```

#### 2.2.2 sessions (会话表)

```sql
CREATE TABLE sessions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id),
    status VARCHAR(20) DEFAULT 'active' COMMENT '状态: active/closed/error',
    context JSONB DEFAULT '{}' COMMENT '会话上下文',
    metadata JSONB DEFAULT '{}' COMMENT '元数据',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    expires_at TIMESTAMP COMMENT '过期时间'
);

-- 索引
CREATE INDEX idx_sessions_user_id ON sessions(user_id);
CREATE INDEX idx_sessions_status ON sessions(status);
CREATE INDEX idx_sessions_expires ON sessions(expires_at);
```

#### 2.2.3 tasks (任务表)

```sql
CREATE TABLE tasks (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    session_id UUID NOT NULL REFERENCES sessions(id),
    parent_task_id UUID REFERENCES tasks(id) COMMENT '父任务ID',
    agent_type VARCHAR(50) NOT NULL COMMENT '智能体类型: coordinator/developer/opener/trader/carer',
    intent VARCHAR(100) COMMENT '意图类型',
    status VARCHAR(20) DEFAULT 'pending' COMMENT '状态: pending/processing/completed/failed/escalated',
    
    -- 输入输出
    input_data JSONB NOT NULL COMMENT '输入数据',
    output_data JSONB COMMENT '输出数据',
    
    -- 底座调用记录
    base_calls JSONB DEFAULT '[]' COMMENT '底座API调用记录',
    
    -- 执行信息
    started_at TIMESTAMP COMMENT '开始时间',
    completed_at TIMESTAMP COMMENT '完成时间',
    error_msg TEXT COMMENT '错误信息',
    
    -- 人工干预
    human_reviewed BOOLEAN DEFAULT FALSE COMMENT '是否人工复核',
    reviewer_id UUID REFERENCES users(id) COMMENT '复核人',
    review_result VARCHAR(20) COMMENT '复核结果: approved/rejected',
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 索引
CREATE INDEX idx_tasks_session_id ON tasks(session_id);
CREATE INDEX idx_tasks_agent_type ON tasks(agent_type);
CREATE INDEX idx_tasks_status ON tasks(status);
CREATE INDEX idx_tasks_created ON tasks(created_at);
```

#### 2.2.4 escalations (请示记录表)

```sql
CREATE TABLE escalations (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    task_id UUID NOT NULL REFERENCES tasks(id),
    escalation_type VARCHAR(50) NOT NULL COMMENT '请示类型: key_customer/major_difficulty/major_risk/important_matter/innovation/major_investment',
    
    -- 请示内容
    title VARCHAR(200) NOT NULL COMMENT '标题',
    content TEXT NOT NULL COMMENT '详细内容',
    evidence JSONB COMMENT '佐证材料',
    
    -- 处理状态
    status VARCHAR(20) DEFAULT 'pending' COMMENT '状态: pending/processing/resolved/rejected',
    priority VARCHAR(20) DEFAULT 'normal' COMMENT '优先级: low/normal/high/urgent',
    
    -- 处理结果
    handler_id UUID REFERENCES users(id) COMMENT '处理人',
    resolution TEXT COMMENT '处理意见',
    resolved_at TIMESTAMP COMMENT '处理时间',
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 索引
CREATE INDEX idx_escalations_task_id ON escalations(task_id);
CREATE INDEX idx_escalations_status ON escalations(status);
CREATE INDEX idx_escalations_type ON escalations(escalation_type);
```

#### 2.2.5 clients (客户表)

```sql
CREATE TABLE clients (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    client_code VARCHAR(50) UNIQUE NOT NULL COMMENT '客户编码',
    name VARCHAR(200) NOT NULL COMMENT '客户名称',
    type VARCHAR(50) NOT NULL COMMENT '类型: private_fund/public_fund/insurance/bank/other',
    scale DECIMAL(15,2) COMMENT '管理规模(万元)',
    
    -- 风险评级
    risk_level VARCHAR(20) DEFAULT 'medium' COMMENT '风险等级: low/medium/high',
    risk_score INTEGER COMMENT '风险评分',
    
    -- 标签
    tags JSONB DEFAULT '[]' COMMENT '标签',
    
    -- 关联信息
    primary_manager_id UUID REFERENCES users(id) COMMENT '主客户经理',
    secondary_manager_id UUID REFERENCES users(id) COMMENT '副客户经理',
    
    -- 底座同步标记
    base_synced BOOLEAN DEFAULT FALSE COMMENT '是否已同步到底座',
    base_synced_at TIMESTAMP COMMENT '同步时间',
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 索引
CREATE INDEX idx_clients_type ON clients(type);
CREATE INDEX idx_clients_risk ON clients(risk_level);
CREATE INDEX idx_clients_manager ON clients(primary_manager_id);
```

#### 2.2.6 metrics (考核指标表)

```sql
CREATE TABLE metrics (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    employee_id UUID NOT NULL REFERENCES users(id),
    metric_type VARCHAR(50) NOT NULL COMMENT '指标类型',
    metric_value DECIMAL(15,4) NOT NULL COMMENT '指标值',
    metric_unit VARCHAR(20) COMMENT '单位',
    
    -- 统计周期
    period_type VARCHAR(20) NOT NULL COMMENT '周期: daily/weekly/monthly/quarterly/yearly',
    period_start DATE NOT NULL COMMENT '周期开始',
    period_end DATE NOT NULL COMMENT '周期结束',
    
    -- 数据源
    source_data JSONB COMMENT '原始数据',
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(employee_id, metric_type, period_type, period_start)
);

-- 索引
CREATE INDEX idx_metrics_employee ON metrics(employee_id);
CREATE INDEX idx_metrics_period ON metrics(period_type, period_start);
```

#### 2.2.7 audit_logs (审计日志表)

```sql
CREATE TABLE audit_logs (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id),
    session_id UUID REFERENCES sessions(id),
    task_id UUID REFERENCES tasks(id),
    
    -- 操作信息
    action VARCHAR(100) NOT NULL COMMENT '操作类型',
    resource VARCHAR(100) NOT NULL COMMENT '操作资源',
    resource_id VARCHAR(100) COMMENT '资源ID',
    
    -- 操作结果
    result VARCHAR(20) NOT NULL COMMENT '结果: success/failure',
    details JSONB COMMENT '详细日志',
    error_code VARCHAR(50) COMMENT '错误码',
    error_msg TEXT COMMENT '错误信息',
    
    -- 客户端信息
    ip_address INET COMMENT 'IP地址',
    user_agent TEXT COMMENT 'User-Agent',
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 索引
CREATE INDEX idx_audit_user ON audit_logs(user_id);
CREATE INDEX idx_audit_action ON audit_logs(action);
CREATE INDEX idx_audit_created ON audit_logs(created_at);
```

## 三、接口详细定义

### 3.1 RESTful API 规范

#### 基础信息

- **Base URL:** `https://api.company.com/institution-broker/v1`
- **协议:** HTTPS
- **认证方式:** Bearer Token (JWT)
- **数据格式:** JSON
- **字符编码:** UTF-8

#### 通用响应格式

```json
{
  "code": 200,
  "message": "success",
  "data": {},
  "trace_id": "uuid-for-tracing"
}
```

#### 错误码定义

| 错误码 | 说明 | HTTP状态码 |
|--------|------|-----------|
| 200 | 成功 | 200 |
| 400 | 请求参数错误 | 400 |
| 401 | 未授权 | 401 |
| 403 | 禁止访问 | 403 |
| 404 | 资源不存在 | 404 |
| 429 | 请求过于频繁 | 429 |
| 500 | 服务器内部错误 | 500 |
| 502 | 底座服务不可用 | 502 |
| 503 | 智能体服务繁忙 | 503 |

### 3.2 核心接口列表

```python
    """意图识别器"""

        'trading_service': 'trading_assistant',
        'customer_service': 'customer_care',

        'research_service': 'forward_research',
        
        # 管理类
        'query_metrics': 'metrics_query',
        'escalate': 'escalation',

        1. 规则匹配（关键词、正则）
        2. 向量相似度匹配（历史意图库）
        3. LLM推理（复杂场景）
```

```python
# src/agents/base_agent.py

from abc import ABC, abstractmethod
from typing import Dict, Any, List
from langchain.schema import SystemMessage, HumanMessage

class BaseAgent(ABC):
    """超级数字员工基类"""
    
    def __init__(
        self,
        role: str,
        goal: str,
        backstory: str,
        base_client: BaseAPIClient,
        llm_client: Any
    ):
        self.role = role
        self.goal = goal
        self.backstory = backstory
        self.base_client = base_client
        self.llm_client = llm_client
        self.tools = []
    
    @property
    @abstractmethod
    def system_prompt(self) -> str:
        """系统提示词（每个子类实现）"""
        pass
    
    async def execute(self, task_input: Dict[str, Any], context: Dict) -> Dict[str, Any]:
        """
        执行任务的标准流程
        """
        # 1. 构建Prompt
        messages = self._build_messages(task_input, context)
        
        # 2. 调用底座获取数据
        base_data = await self._fetch_base_data(task_input)
        
        # 3. 工具调用（如有）
        tool_results = await self._execute_tools(task_input)
        
        # 4. LLM推理
        response = await self.llm_client.chat(
            messages=messages,
            context={
                'base_data': base_data,
                'tool_results': tool_results
            }
        )
        
        # 5. 结果后处理
        result = self._process_response(response)
        
        # 6. 检查是否需要人工确认
        requires_confirmation = self._check_confirmation_needed(result)
        
        return {
            'success': True,
            'result': result,
            'requires_confirmation': requires_confirmation,
            'confirmation_items': self._extract_confirmation_items(result) if requires_confirmation else [],
            'base_calls': self._get_base_call_log(),
            'suggested_next': self._suggest_next_action(result)
        }
    
    def _build_messages(self, task_input: Dict, context: Dict) -> List:
        """构建消息列表"""
        messages = [
            SystemMessage(content=self.system_prompt),
            HumanMessage(content=self._format_task(task_input))
        ]
        return messages
    
    @abstractmethod
    async def _fetch_base_data(self, task_input: Dict) -> Dict:
        """获取底座数据（子类实现）"""
        pass
    
    def _check_confirmation_needed(self, result: Dict) -> bool:
        """检查是否需要人工确认"""
        # 检查涉及资金的场景
        if result.get('involves_fund', False):
            return True
        # 检查高风险场景
        if result.get('risk_level') == 'high':
            return True
        # 检查大额交易
        if result.get('amount', 0) > 5_000_000:
            return True
        return False
```

```python
# src/security/data_masking.py

import re

class DataMasker:
    """数据脱敏器"""
    
    @staticmethod
    def mask_phone(phone: str) -> str:
        """手机号脱敏"""
        return re.sub(r'(\d{3})\d{4}(\d{4})', r'\1****\2', phone)
    
    @staticmethod
    def mask_id_card(id_card: str) -> str:
        """身份证号脱敏"""
        return re.sub(r'(\d{6})\d{8}(\d{4})', r'\1********\2', id_card)
    
    @staticmethod
    def mask_account_no(account_no: str) -> str:
        """账号脱敏"""
        if len(account_no) > 8:
            return account_no[:4] + '*' * (len(account_no) - 8) + account_no[-4:]
        return account_no
    
    @staticmethod
    def mask_sensitive_data(data: dict) -> dict:
        """批量脱敏"""
        masked = data.copy()
        sensitive_fields = ['phone', 'mobile', 'id_card', 'id_no', 'account_no', 'bank_account']
        
        for field in sensitive_fields:
            if field in masked:
                value = masked[field]
                if 'phone' in field or 'mobile' in field:
                    masked[field] = DataMasker.mask_phone(value)
                elif 'id' in field:
                    masked[field] = DataMasker.mask_id_card(value)
                elif 'account' in field:
                    masked[field] = DataMasker.mask_account_no(value)
        
        return masked
```

```python
from fastapi import Request, FastAPI
from fastapi.responses import JSONResponse

async def global_exception_handler(request: Request, exc: Exception):
    """全局异常处理器"""
    
    if isinstance(exc, BaseAPIError):
        return JSONResponse(
            status_code=exc.http_status,
            content={
                'code': exc.error_code.value,
                'message': exc.message,
                'data': exc.data,
                'trace_id': get_trace_id()
            }
        )
    
    elif isinstance(exc, ValidationError):
        return JSONResponse(
            status_code=400,
            content={
                'code': ErrorCode.BAD_REQUEST.value,
                'message': '请求参数验证失败',
                'data': {'errors': exc.errors()},
                'trace_id': get_trace_id()
            }
        )
    
    # 未知错误
    logger.error(f"Unhandled exception: {exc}", exc_info=True)
    return JSONResponse(
        status_code=500,
        content={
            'code': ErrorCode.INTERNAL_ERROR.value,
            'message': '服务器内部错误',
            'data': {},
            'trace_id': get_trace_id()
        }
    )

def register_exception_handlers(app: FastAPI):
    """注册异常处理器"""
    app.add_exception_handler(Exception, global_exception_handler)
    app.add_exception_handler(BaseAPIError, global_exception_handler)
```
