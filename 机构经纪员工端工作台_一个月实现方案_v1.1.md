## 一、方案概述

### 1.1 核心思路
- **自动化功能**：采用OpenClaw方式实现（客户查重、表单预审、AI画像生成等）
- **智能功能**：采用智能体调用方式（复杂咨询、智能推荐、请示内容生成等）
- **前端入口**：保持Web H5方式，与原型设计保持一致

### 1.2 技术选型
| 层级 | 技术方案 |
|------|----------|
| 智能体编排 | OpenClaw Gateway |
| 前端框架 | Vue3 + Vite + TypeScript |
| 后端服务 | Node.js / Python FastAPI |
| 数据库 | PostgreSQL + Redis |
| 部署 | Docker Compose |

## 四、自动化 vs 智能体划分

| 功能场景 | 自动化 (OpenClaw) | 智能体调用 |
|---------|------------------|-----------|
| **客户查重** | ✓ 自动比对数据库，返回结果 | - |
| **AI画像生成** | ✓ 调用底座关系网络自动构建 | - |
| **表单预审** | ✓ 规则引擎自动检查 | - |
| **进度状态同步** | ✓ 定时任务自动更新 | - |
| **智能推荐** | - | ✓ 大模型推理后返回 |
| **复杂咨询** | - | ✓ 多轮对话理解意图 |
| **请示审批** | ✓ 流程自动流转 | ✓ 内容智能生成 |

### 7.2 客户查重 Skill

#### 7.2.1 SKILL.md

```markdown
# customer-duplicate-check

## 描述
自动查询客户是否已存在于系统中，基于名称和统一社会信用代码进行模糊匹配，返回相似度分析结果。

## 输入参数

| 参数名 | 类型 | 必填 | 说明 |
|--------|------|------|------|
| customer_name | string | 是 | 客户名称 |
| unified_code | string | 否 | 统一社会信用代码 |
| fuzzy_match | boolean | 否 | 是否启用模糊匹配，默认true |

## 输出结果

```json
{
  "is_duplicate": false,
  "confidence": 0.85,
  "similar_customers": [
    {
      "customer_id": "C20240306001",
      "customer_name": "XX投资管理有限公司",
      "unified_code": "91110000XXXXXXXX",
      "similarity": 0.92,
      "owner_name": "张三",
      "status": "signed"
    }
  ],
  "suggestion": "建议进一步核实是否为同一客户"
}
```

## 匹配规则

1. **精确匹配**：统一社会信用代码完全一致 → 置信度1.0
2. **强相似**：名称相似度>0.9 → 置信度0.8-0.95
3. **弱相似**：名称相似度0.7-0.9 → 置信度0.5-0.8
4. **不相似**：名称相似度<0.7 → 置信度<0.5
```

#### 7.2.2 index.js

```javascript
const { Client } = require('pg');
const { similarity } = require('string-similarity');

module.exports = async function(params, context) {
  const { customer_name, unified_code, fuzzy_match = true } = params;
  
  const client = new Client({
    connectionString: process.env.DATABASE_URL
  });
  await client.connect();
  
  try {
    let similarCustomers = [];
    let isDuplicate = false;
    let confidence = 0;
    
    // 1. 精确匹配统一社会信用代码
    if (unified_code) {
      const exactMatch = await client.query(
        'SELECT * FROM customers WHERE unified_code = $1',
        [unified_code]
      );
      if (exactMatch.rows.length > 0) {
        isDuplicate = true;
        confidence = 1.0;
        similarCustomers = exactMatch.rows.map(row => ({
          customer_id: row.customer_no,
          customer_name: row.customer_name,
          unified_code: row.unified_code,
          similarity: 1.0,
          owner_name: row.owner_name,
          status: row.status
        }));
      }
    }
    
    // 2. 模糊匹配客户名称
    if (!isDuplicate && fuzzy_match) {
      // 获取所有客户名称进行相似度计算
      const allCustomers = await client.query(
        'SELECT customer_no, customer_name, unified_code, owner_name, status FROM customers'
      );
      
      for (const row of allCustomers.rows) {
        const nameSimilarity = similarity(customer_name, row.customer_name);
        if (nameSimilarity > 0.6) {
          similarCustomers.push({
            customer_id: row.customer_no,
            customer_name: row.customer_name,
            unified_code: row.unified_code,
            similarity: nameSimilarity,
            owner_name: row.owner_name,
            status: row.status
          });
        }
      }
      
      // 按相似度排序，取前5
      similarCustomers.sort((a, b) => b.similarity - a.similarity);
      similarCustomers = similarCustomers.slice(0, 5);
      
      if (similarCustomers.length > 0) {
        confidence = similarCustomers[0].similarity;
        if (confidence > 0.9) isDuplicate = true;
      }
    }
    
    // 3. 调用底座关系网络增强匹配
    if (context.baseAPI) {
      try {
        const baseResult = await context.baseAPI.query('entity_search', {
          entity_type: 'organization',
          name: customer_name,
          unified_code: unified_code
        });
        
        if (baseResult && baseResult.length > 0) {
          // 合并底座返回的结果
          similarCustomers = mergeResults(similarCustomers, baseResult);
        }
      } catch (e) {
        console.log('底座查询失败，使用本地结果:', e.message);
      }
    }
    
    // 生成建议
    let suggestion = '';
    if (isDuplicate) {
      suggestion = '该客户已存在于系统中，请勿重复录入。可联系归属人' + 
                   (similarCustomers[0]?.owner_name || '') + '协同跟进。';
    } else if (confidence > 0.8) {
      suggestion = '发现高度相似客户，建议进一步核实是否为同一客户。';
    } else if (confidence > 0.5) {
      suggestion = '发现部分相似客户，建议人工确认。';
    } else {
      suggestion = '未发现重复客户，可以继续录入。';
    }
    
    return {
      is_duplicate: isDuplicate,
      confidence: confidence,
      similar_customers: similarCustomers,
      suggestion: suggestion
    };
    
  } finally {
    await client.end();
  }
};

function mergeResults(localResults, baseResults) {
  // 合并本地和底座返回的结果，去重
  const merged = [...localResults];
  for (const baseItem of baseResults) {
    const exists = merged.find(m => m.customer_id === baseItem.id);
    if (!exists) {
      merged.push({
        customer_id: baseItem.id,
        customer_name: baseItem.name,
        unified_code: baseItem.unified_code,
        similarity: baseItem.confidence || 0.7,
        source: 'base'
      });
    }
  }
  return merged.sort((a, b) => b.similarity - a.similarity).slice(0, 5);
}

function categorizeScale(capital) {
  if (!capital) return '未知';

  return '小型';
}

function calculateYears(date) {
  if (!date) return null;
  return Math.floor((Date.now() - new Date(date).getTime()) / (365 * 24 * 60 * 60 * 1000));
}

function categorizeFrequency(tradeCount) {
  if (tradeCount > 100) return '高频';
  if (tradeCount > 20) return '中高频';
  if (tradeCount > 5) return '中频';
  return '低频';
}

function categorizeOrderSize(avgVolume) {
  if (!avgVolume) return '未知';
  if (avgVolume >= 50000000) return '5000万+';
  if (avgVolume >= 10000000) return '1000万-5000万';
  if (avgVolume >= 1000000) return '100万-1000万';
  return '100万以下';
}

function calculateDays(date) {
  if (!date) return null;

const RULES = require('./rules.json');

module.exports = async function(params, context) {
  const { form_data, account_type } = params;
  
  const result = {
    check_result: 'pass', // pass / conditional_pass / fail
    score: 100,
    missing_items: [],
    risk_alerts: [],
    suggestions: [],
    auto_fill_suggestions: {}
  };
  
  try {
    // 1. 必填项检查
    const requiredCheck = checkRequiredFields(form_data, account_type);
    result.missing_items = requiredCheck.missing;
    result.score -= requiredCheck.missing.length * 10;
    
    // 2. 格式校验
    const formatCheck = checkFormat(form_data);
    if (formatCheck.errors.length > 0) {
      result.risk_alerts.push(...formatCheck.errors.map(e => ({
        level: 'error',
        type: 'format',
        message: e
      })));
      result.score -= formatCheck.errors.length * 5;
    }
    
    // 3. 黑名单检查
    const blacklistCheck = await checkBlacklist(form_data, context);
    if (blacklistCheck.hit) {
      result.risk_alerts.push({
        level: 'critical',
        type: 'blacklist',
        message: blacklistCheck.reason
      });
      result.score -= 50;
    }
    
    // 4. 反洗钱检查
    const amlCheck = await checkAML(form_data, context);
    if (amlCheck.risk_level === 'high') {
      result.risk_alerts.push({
        level: 'warning',
        type: 'aml',
        message: '反洗钱风险等级较高，需人工复核'
      });
      result.score -= 20;
    }
    
    // 5. 适当性匹配
    const suitabilityCheck = await checkSuitability(form_data, context);
    if (!suitabilityCheck.matched) {
      result.risk_alerts.push({
        level: 'warning',
        type: 'suitability',
        message: suitabilityCheck.reason
      });
      result.score -= 15;
    }
    
    // 6. 关联交易检查
    const relatedPartyCheck = await checkRelatedParty(form_data, context);
    if (relatedPartyCheck.has_related) {
      result.risk_alerts.push({
        level: 'info',
        type: 'related_party',
        message: `发现${relatedPartyCheck.count}个关联方`
      });
    }
    
    // 7. 智能填充建议
    result.auto_fill_suggestions = await generateAutoFill(form_data, context);
    
    // 8. 生成建议
    result.suggestions = generateSuggestions(result);
    
    // 9. 确定最终结果
    if (result.score < 50 || result.risk_alerts.some(r => r.level === 'critical')) {
      result.check_result = 'fail';
    } else if (result.score < 80 || result.missing_items.length > 0) {
      result.check_result = 'conditional_pass';
    }
    
    return result;
    
  } catch (error) {
    console.error('预审失败:', error);
    return {
      check_result: 'fail',
      score: 0,
      missing_items: [],
      risk_alerts: [{
        level: 'error',
        type: 'system',
        message: '预审系统异常，请人工审核'
      }],
      suggestions: ['请联系技术支持'],
      auto_fill_suggestions: {}
    };
  }
};

// 必填项检查
function checkRequiredFields(formData, accountType) {
  const requiredFields = {
    '自营': ['company_name', 'unified_code', 'legal_person', 'business_license', 'risk_assessment'],
    '资管': ['manager_name', 'manager_license', 'product_scope', 'risk_disclosure'],
    '产品': ['product_code', 'product_name', 'product_contract', 'custody_agreement', 'product_registration']
  };
  
  const fields = requiredFields[accountType] || [];
  const missing = [];
  
  for (const field of fields) {
    if (!formData[field] || formData[field].trim() === '') {
      missing.push(field);
    }
  }
  
  return { missing };
}

// 格式校验
function checkFormat(formData) {
  const errors = [];
  
  // 统一社会信用代码格式
  if (formData.unified_code && !/^\w{18}$/.test(formData.unified_code)) {
    errors.push('统一社会信用代码格式不正确');
  }
  
  // 手机号格式
  if (formData.contact_phone && !/^1[3-9]\d{9}$/.test(formData.contact_phone)) {
    errors.push('联系人手机号格式不正确');
  }
  
  // 邮箱格式
  if (formData.contact_email && !/^.+@.+\..+$/.test(formData.contact_email)) {
    errors.push('邮箱格式不正确');
  }
  
  return { errors };
}

// 黑名单检查
async function checkBlacklist(formData, context) {
  const db = context.db;
  
  // 检查客户名称
  const nameCheck = await db.query(
    'SELECT 1 FROM blacklist WHERE name = $1 AND type = "sanction"',
    [formData.company_name]
  );
  
  if (nameCheck.rows.length > 0) {
    return { hit: true, reason: '客户名称命中制裁名单' };
  }
  
  // 检查统一信用代码
  if (formData.unified_code) {
    const codeCheck = await db.query(
      'SELECT 1 FROM blacklist WHERE unified_code = $1',
      [formData.unified_code]
    );
    
    if (codeCheck.rows.length > 0) {
      return { hit: true, reason: '统一社会信用代码命中黑名单' };
    }
  }
  
  return { hit: false };
}

async function checkAML(formData, context) {

// 适当性匹配
async function checkSuitability(formData, context) {
  const customerRisk = formData.risk_level;
  const productRisk = formData.product_risk_level;
  
  if (!customerRisk || !productRisk) {
    return { matched: false, reason: '缺少风险等级信息' };
  }
  
  const riskMap = { 'R1': 1, 'R2': 2, 'R3': 3, 'R4': 4, 'R5': 5 };
  
  if (riskMap[customerRisk] < riskMap[productRisk]) {
    return { 
      matched: false, 
      reason: `客户风险承受能力(${customerRisk})低于产品风险等级(${productRisk})` 
    };
  }
  
  return { matched: true };
}

async function checkRelatedParty(formData, context) {

        unified_code: formData.unified_code

      return {
        has_related: relations && relations.length > 0,
        count: relations ? relations.length : 0,
        parties: relations || []
      };

      return { has_related: false, count: 0 };
    }

  return { has_related: false, count: 0 };

// 生成智能填充建议
async function generateAutoFill(formData, context) {
  const suggestions = {};
  
  // 基于统一信用代码推断行业
  if (formData.unified_code && !formData.industry_code) {
    const industryMap = {
      '11': 'A01', // 农业
      '51': 'F51', // 批发
      '66': 'J66', // 货币金融
      '67': 'J67', // 资本市场
      // ... 更多映射
    };
    
    const code = formData.unified_code.substring(0, 2);
    if (industryMap[code]) {
      suggestions.industry_code = industryMap[code];
    }
  }
  
  // 基于账户类型建议风险等级
  if (!formData.risk_level) {
    const riskMap = {
      '自营': 'R3',
      '资管': 'R4',
      '产品': 'R4'
    };
    suggestions.risk_level = riskMap[formData.account_type] || 'R3';
  }
  
  return suggestions;
}

// 生成建议
function generateSuggestions(result) {
  const suggestions = [];
  
  if (result.missing_items.length > 0) {
    suggestions.push(`补充以下材料：${result.missing_items.join('、')}`);
  }
  
  for (const alert of result.risk_alerts) {
    if (alert.level === 'warning' && alert.type === 'compliance') {
      suggestions.push(alert.message);
    }
  }
  
  return suggestions;
}

### 8.2 详细规则设计

#### 8.2.1 客户查重规则

```yaml
规则名称: 客户查重判定
触发条件: 客户开发表单提交时自动触发

匹配算法:
  精确匹配:
    条件: 统一社会信用代码完全一致
    结果: 置信度1.0，标记为重复，禁止录入
    
  模糊匹配:
    算法: 编辑距离 + Jaccard相似度
    阈值设置:
      高相似(0.9-1.0):  提示"高度疑似重复，请核实"
      中相似(0.7-0.9):  提示"发现相似客户，建议确认"
      低相似(0.5-0.7):  展示相似列表供参考
      不相似(<0.5):     正常录入

底座增强:
  调用: entity_search接口
  补充: 别名、历史名称、关联实体
  权重: 底座返回结果权重+0.1

输出:
  - is_duplicate: 是否重复(布尔)
  - confidence: 置信度(0-1)
  - similar_customers: 相似客户列表(最多5个)
  - suggestion: 处理建议文本
```

#### 8.2.2 客户分配规则

```yaml
规则名称: 智能客户分配
触发条件: 新客户录入且无指定归属时

分配算法:
  步骤1_负载均衡:
    计算: 当前待跟进客户数 / 月度目标客户数
    权重: 40%
    
  步骤2_能力匹配:
    维度:
      - 行业专长匹配度(客户行业 vs 员工历史成交行业)
      - 客户规模匹配度(客户规模 vs 员工擅长规模)
    权重: 35%
    
  步骤3_关系延续:
    检查: 是否与客户有过历史接触
    加分: 有历史接触+20分
    权重: 25%

特殊规则:
  战略客户: 直接分配给团队长
  黑名单关联: 标记提醒
  冲突回避: 3个月内不得分配给同一客户

输出:
  - assigned_to: 分配员工ID
  - reason: 分配理由
  - confidence: 分配置信度
```

#### 8.2.3 开户预审规则

```yaml
规则名称: 开户资料预审
触发条件: 开户申请提交时

检查流程:
  阶段1_完整性检查:
    规则:
      自营户: 必填[营业执照,法人身份证,授权书,风险测评]
      资管户: 必填[管理人资格,产品备案,投资范围说明]
      产品户: 必填[产品合同,托管协议,备案证明]
    扣分: 每项缺失-10分

  阶段2_格式校验:
    规则:
      统一代码: 18位字母数字
      手机号: 1[3-9]开头11位
      邮箱: 包含@和域名
    扣分: 每项错误-5分

  阶段3_黑名单检查:
    数据源:
      - 制裁名单(OFAC,UN)
      - 反洗钱高风险名单
      - 公司内部黑名单
    处理: 命中即拒绝，扣分-50

  阶段4_反洗钱评级:
    接口: 风控系统aml_risk
    分级:
      低风险: 不扣分
      中风险: -10分
      高风险: -20分，需人工复核

  阶段5_适当性匹配:
    规则: 客户风险等级 >= 产品风险等级
    不匹配: -15分

  阶段6_关联交易检查:
    调用: 底座related_parties
    处理: 发现关联方标记提醒，不扣分

评分标准:
  90-100分: 通过
  60-89分:  有条件通过(需补充材料)
  <60分:    不通过
```

#### 8.2.4 机器人对话规则

```yaml
规则名称: 对话意图识别与响应
触发条件: 用户发送消息

意图识别:
  层级1_关键词匹配:
    "我的客户" → 客户列表查询
    "查一下" + 客户名 → 客户详情查询
    "开户进度" → 开户状态查询
    "帮我推进" → 任务推进
    
  层级2_语义理解(大模型):
    模型: kimi-coding/k2p5
    提示词: |
      识别用户意图，从以下选项中选择：
      - query_customer: 查询客户信息
      - query_task: 查询待办任务
      - query_performance: 查询业绩
      - create_task: 创建任务
      - generate_plan: 生成方案
      - general_chat: 闲聊
      
      用户输入: {user_input}
      
      返回JSON格式：
      {
        "intent": "意图类型",
        "entities": {
          "customer_name": "客户名",
          "date_range": "日期范围"
        },
        "confidence": 0.95
      }

响应生成:
  数据查询类:
    - 解析参数 → 查询数据库 → 封装卡片 → 返回
  
  任务操作类:
    - 确认意图 → 执行操作 → 反馈结果
  
  内容生成类:
    - 收集信息 → 调用大模型 → 润色返回

多轮对话:
  状态保持: session_id关联上下文
  槽位填充: 缺失参数追问
  确认机制: 高风险操作二次确认
```

#### 8.2.5 请示生成规则

```yaml
规则名称: AI请示内容生成
触发条件: 用户选择请示类型并填写基础信息

生成流程:
  步骤1_模板选择:
    映射:
      战略客户政策 → template_strategic_client
      资源投入申请 → template_resource_request
      合规风险请示 → template_compliance_risk
      横向协同-研究所 → template_collab_research
      横向协同-资管 → template_collab_asset

  步骤2_信息收集:
    必填:
      - 客户名称
      - 请示事由
      - 期望支持
    选填:
      - 背景说明
      - 预期效果
      - 时间节点

  步骤3_模板填充:
    使用Handlebars语法填充基础信息

  步骤4_大模型润色:
    模型: kimi-coding/k2p5
    提示词: |
      基于以下信息生成正式请示内容：
      
      模板: {template}
      基础信息: {data}
      
      要求：
      1. 使用正式公文语气
      2. 逻辑清晰，理由充分
      3. 突出业务价值和紧迫性

  步骤5_人工确认:
    展示生成内容 → 用户编辑/确认 → 提交审批

审批路由:
  向上请示: 提交人 → 直属领导 → 部门负责人 → 公司领导
  横向协同: 提交人 → 本部门领导 → 协同部门领导 → 协同部门经办人
